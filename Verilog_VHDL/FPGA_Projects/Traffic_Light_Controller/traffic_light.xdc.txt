## Clock Signal (50 MHz)
set_property -dict { PACKAGE_PIN N11   IOSTANDARD LVCMOS33 } [get_ports { clk }];
create_clock -period 20.000 -name sys_clk_pin -waveform {0.000 10.000} -add [get_ports { clk }];

## Switches (Car Sensors)
set_property -dict { PACKAGE_PIN L5    IOSTANDARD LVCMOS33 } [get_ports { Sa }]; # SW0
set_property -dict { PACKAGE_PIN L4    IOSTANDARD LVCMOS33 } [get_ports { Sb }]; # SW1

## LEDs (Road A Lights)
set_property -dict { PACKAGE_PIN J3    IOSTANDARD LVCMOS33 } [get_ports { Ra }]; # LED0 (Red A)
set_property -dict { PACKAGE_PIN H3    IOSTANDARD LVCMOS33 } [get_ports { Ya }]; # LED1 (Yellow A)
set_property -dict { PACKAGE_PIN J1    IOSTANDARD LVCMOS33 } [get_ports { Ga }]; # LED2 (Green A)

## LEDs (Road B Lights)
set_property -dict { PACKAGE_PIN K1    IOSTANDARD LVCMOS33 } [get_ports { Rb }]; # LED3 (Red B)
set_property -dict { PACKAGE_PIN L3    IOSTANDARD LVCMOS33 } [get_ports { Yb }]; # LED4 (Yellow B)
set_property -dict { PACKAGE_PIN L2    IOSTANDARD LVCMOS33 } [get_ports { Gb }]; # LED5 (Green B)
